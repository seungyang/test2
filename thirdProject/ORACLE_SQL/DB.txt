-- 회원정보 테이블
CREATE TABLE CUSTOMER_INFO
(
CUSTOMER_ID VARCHAR2(30) NOT NULL, 
CUSTOMER_PASSWORD VARCHAR2(30) NOT NULL, 
CUSTOMER_NAME VARCHAR2(30) NOT NULL, 
CUSTOMER_GENDER VARCHAR2(10) NOT NULL, 
CUSTOMER_PH VARCHAR2(20) NOT NULL, 
CUSTOMER_EMAIL VARCHAR2(50) NOT NULL, 
CUSTOMER_AGE NUMBER(3) NOT NULL, 
CUSTOMER_FLAG VARCHAR2(3) NOT NULL,
CONSTRAINT PK_CUSTOMER_INFO PRIMARY KEY(CUSTOMER_ID)
);



-- 업체정보 테이블
CREATE TABLE COMPANY_INFO
(
COMPANY_ID VARCHAR2(30) NOT NULL,
COMPANY_PASSWORD VARCHAR2(30) NOT NULL,
COMPANY_NAME VARCHAR2(30) NOT NULL,
COMPANY_PH VARCHAR2(20) NOT NULL,
COMPANY_EMAIL VARCHAR2(50) NOT NULL,
COMPANY_REGNUM VARCHAR2(15) NOT NULL,
COMPANY_FLAG VARCHAR2(3) NOT NULL,
CONSTRAINT PK_COMPANY_INFO PRIMARY KEY(COMPANY_ID)
);



--전시회 아이디에 쓸 시퀀스 생성
CREATE SEQUENCE SEQ_EXHIBITION
	START WITH 1
	INCREMENT BY 1
	MAXVALUE 10000
	MINVALUE 1
	NOCYCLE;

-- 전시회 정보 테이블
CREATE TABLE EXHIBITION_INFO
(
EXHIBITION_ID NUMBER(5) NOT NULL,
EXHIBITION_TITLE VARCHAR2(300) NOT NULL,
EXHIBITION_START_DATE VARCHAR2(20) NOT NULL,
EXHIBITION_END_DATE VARCHAR2(20) NOT NULL,
EXHIBITION_TOTAL_DATE VARCHAR2(50) NOT NULL,
EXHIBITION_IMAGE VARCHAR2(2000) NOT NULL,
EXHIBITION_CONTENTS VARCHAR2(2000) NOT NULL,
EXHIBITION_LOCATION VARCHAR2(1000) NOT NULL,
EXHIBITION_PRICE NUMBER(10) NOT NULL,
EXHIBITION_LATITUDE VARCHAR2(5),
EXHIBITION_LONGITUDE VARCHAR2(5),
CONSTRAINT PK_EXHIBITION_INFO PRIMARY KEY(EXHIBITION_ID)
);




--일반회원 예매 테이블 시퀀스 생성
CREATE SEQUENCE SEQ_TICKETING
	START WITH 1
	INCREMENT BY 1
	MAXVALUE 1000000
	MINVALUE 1
	NOCYCLE;

-- 일반회원 예매내역 테이블
CREATE TABLE TICKETING
(
TICKETING_ID NUMBER(7) NOT NULL,
TICKETING_DATE VARCHAR2(20) NOT NULL,
CONSUMER_PH VARCHAR2(20) NOT NULL,
CONSTRAINT PK_TICKETING PRIMARY KEY(TICKETING_ID)
);




--일반회원 찜목록 테이블 시퀀스 생성
CREATE SEQUENCE SEQ_WISH_LIST
	START WITH 1
	INCREMENT BY 1
	MAXVALUE 1000000
	MINVALUE 1
	NOCYCLE;
	
-- 일반회원 찜목록 테이블
CREATE TABLE WISH_LIST
(
WISH_LIST_ID NUMBER(7) NOT NULL,
WISH_LIST_FLAG VARCHAR2(3) NOT NULL,
CONSTRAINT PK_WISH_LIST PRIMARY KEY(WISH_LIST_ID)
);




--일반회원 후기 테이블 시퀀스 생성
CREATE SEQUENCE SEQ_REVIEW
	START WITH 1
	INCREMENT BY 1
	MAXVALUE 1000000
	MINVALUE 1
	NOCYCLE;
	
-- 후기 테이블 생성
CREATE TABLE REVIEW
(
REVIEW_ID NUMBER(7) NOT NULL,
REVIEW_TITLE VARCHAR2(300) NOT NULL,
REVIEW_CONTENTS VARCHAR2(2000) NOT NULL,
REVIEW_DATE VARCHAR2(20) NOT NULL,
REVIEW_STAR_SCORE NUMBER(2,1) NOT NULL, --NUMBER(2,1) 숫자2개 소수점1자리 ex)1.5, 4.5
REVIEW_IMAGE VARCHAR2(2000),
CONSTRAINT PK_REVIEW PRIMARY KEY(REVIEW_ID)
);
